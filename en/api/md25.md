# Il2Cpp (frida-il2cpp-bridge)

JsHook bundles [frida-il2cpp-bridge](https://github.com/vfsfitvnm/frida-il2cpp-bridge) v0.13.0, exposing the `globalThis.Il2Cpp` namespace for runtime reverse engineering of Unity Il2Cpp games and applications.

Unity 5.3.0 through 6000.3.x are supported. This API only works in Unity applications built with the Il2Cpp backend.

---

## Quick Start

```javascript
Il2Cpp.perform(() => {
    // Il2Cpp is initialized; all APIs are now safe to access
    console.log("Unity version:", Il2Cpp.unityVersion);
    console.log("App identifier:", Il2Cpp.application.identifier);
    console.log("App version:", Il2Cpp.application.version);

    // Iterate over the assemblies
    for (const assembly of Il2Cpp.domain.assemblies) {
        console.log("Assembly:", assembly.name);
    }
});
```

> `Il2Cpp.perform()` is the entry point for everything. It automatically waits for `libil2cpp.so` to load and finish initializing, then attaches the current thread to the Il2Cpp domain.

---

## Configuration

Use `Il2Cpp.$config` to override auto-detected values **before** calling `perform()`:

```javascript
// Override the module name (defaults to libil2cpp.so on Android)
Il2Cpp.$config.moduleName = "libil2cpp.so";

// Override the Unity version
Il2Cpp.$config.unityVersion = "2021.3.0f1";

// Override export addresses (to defeat obfuscation / stripping)
Il2Cpp.$config.exports = {
    il2cpp_class_from_name: () => Il2Cpp.module.base.add(0x1204c),
};

Il2Cpp.perform(() => { /* ... */ });
```

## Core Properties

| Property | Type | Description |
|----------|------|-------------|
| `Il2Cpp.module` | `Module` | The Il2Cpp native module (libil2cpp.so) |
| `Il2Cpp.domain` | `Il2Cpp.Domain` | The application domain |
| `Il2Cpp.unityVersion` | `string` | The Unity engine version |
| `Il2Cpp.application.identifier` | `string` | The app's package name |
| `Il2Cpp.application.version` | `string` | The app's version |
| `Il2Cpp.application.dataPath` | `string` | The app's data path |
| `Il2Cpp.corlib` | `Il2Cpp.Image` | The core library (mscorlib) |

## Class

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");

    // Basic info
    console.log("Name:", klass.name);
    console.log("Namespace:", klass.namespace);
    console.log("Full name:", klass.fullName);
    console.log("Instance size:", klass.instanceSize);
    console.log("Parent:", klass.parent?.fullName);

    // Type checks
    console.log("Is abstract:", klass.isAbstract);
    console.log("Is enum:", klass.isEnum);
    console.log("Is interface:", klass.isInterface);
    console.log("Is generic:", klass.isGeneric);
    console.log("Is value type:", klass.isValueType);

    // Fields
    for (const field of klass.fields) {
        console.log(`  Field: ${field.name} (${field.type.name}) offset=0x${field.offset.toString(16)}`);
    }

    // Methods
    for (const method of klass.methods) {
        console.log(`  Method: ${method.name}(${method.parameterCount} params) -> ${method.returnType.name}`);
    }

    // Nested classes
    for (const nested of klass.nestedClasses) {
        console.log(`  Nested class: ${nested.name}`);
    }

    // Interfaces
    for (const iface of klass.interfaces) {
        console.log(`  Interface: ${iface.fullName}`);
    }

    // Inheritance chain
    for (const parent of klass.hierarchy()) {
        console.log(`  Inherits: ${parent.fullName}`);
    }

    // Get a field/method by name
    const hpField = klass.field("hp");
    const attackMethod = klass.method("Attack", 1); // 1 parameter

    // Run the static constructor
    klass.initialize();

    // Create a new object
    const newObj = klass.new();

    // Enumerate every defined class
    Il2Cpp.Class.enumerate((klass) => {
        // Callback for each class
    });
});
```

## Method

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");

    const method = klass.method("TakeDamage", 1);

    // Basic info
    console.log("Name:", method.name);
    console.log("Parameter count:", method.parameterCount);
    console.log("Return type:", method.returnType.name);
    console.log("Is static:", method.isStatic);
    console.log("Access modifier:", method.modifier);
    console.log("Virtual address:", method.virtualAddress);
    console.log("RVA:", method.relativeVirtualAddress);

    // Parameter info
    for (const param of method.parameters) {
        console.log(`  Parameter: ${param.name} (${param.type.name})`);
    }

    // Invoke a static method
    const staticMethod = klass.method("GetInstance");
    const result = staticMethod.invoke();

    // Replace the method implementation
    method.implementation = function (damage) {
        console.log("TakeDamage called, damage =", damage);
        // Return the original behavior or a custom value
        return;
    };

    // Restore the original implementation
    method.revert();

    // Intercept the method (observe only, without replacing)
    method.intercept({
        onEnter(instance, parameters) {
            console.log("Enter TakeDamage, this =", instance, "damage =", parameters.damage);
        },
        onLeave(returnValue) {
            console.log("Leave TakeDamage, return =", returnValue);
        }
    });

    // Instantiate a generic method
    const genericMethod = klass.method("DoSomething");
    const inflated = genericMethod.inflate(Il2Cpp.corlib.class("System.Int32"));
});
```

## Object

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");

    // Search the heap for objects via the GC
    const objects = Il2Cpp.gc.choose(klass);
    for (const obj of objects) {
        // Instance field — accessed via the object
        const hp = obj.field("hp");
        console.log("Current HP:", hp.value);
        hp.value = 99999;

        // Call an instance method
        const name = obj.method("get_Name").invoke();
        console.log("Player name:", name);

        // Access base-class members
        const baseObj = obj.base;
        console.log("Base class:", baseObj.class.fullName);
    }

    // Create a new object
    const newObj = klass.new();

    // Allocate without calling the constructor
    const allocated = klass.alloc();
});
```

## Trace (Method Tracing)

```javascript
Il2Cpp.perform(() => {
    // Trace methods across an entire assembly
    Il2Cpp.trace()
        .assemblies(Il2Cpp.domain.assembly("Assembly-CSharp"))
        .filterClasses(klass => klass.namespace === "Game")
        .filterMethods(method => !method.isStatic)
        .and()
        .attach();

    // Trace every method of a specific class
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");
    Il2Cpp.trace()
        .classes(klass)
        .and()
        .attach();

    // Trace a specific method (including argument values)
    Il2Cpp.trace(true) // true = log arguments
        .methods(klass.method("TakeDamage"))
        .and()
        .attach();

    // Trace with a call stack
    Il2Cpp.backtrace()
        .methods(klass.method("TakeDamage"))
        .and()
        .attach();

    // Verbose mode (no deduplication)
    Il2Cpp.trace()
        .verbose(true)
        .domain()
        .filterAssemblies(a => a.name === "Assembly-CSharp")
        .filterClasses(c => c.fullName.startsWith("Game."))
        .and()
        .attach();
});
```

## GC (Garbage Collector)

```javascript
Il2Cpp.perform(() => {
    // GC info
    console.log("Heap size:", Il2Cpp.gc.heapSize);
    console.log("Used:", Il2Cpp.gc.usedHeapSize);
    console.log("Is enabled:", Il2Cpp.gc.isEnabled);
    console.log("Is incremental:", Il2Cpp.gc.isIncremental);

    // Disable / enable the GC
    Il2Cpp.gc.isEnabled = false;
    Il2Cpp.gc.isEnabled = true;

    // Search the heap for instances of a class
    const playerClass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.Player");
    const players = Il2Cpp.gc.choose(playerClass);
    console.log("Found", players.length, "Player instances");
});
```

## Memory Snapshot

```javascript
Il2Cpp.perform(() => {
    // Use a memory snapshot (freed automatically)
    Il2Cpp.memorySnapshot(snapshot => {
        // All initialized classes
        for (const klass of snapshot.classes) {
            console.log(klass.fullName);
        }

        // All tracked objects
        for (const obj of snapshot.objects) {
            console.log(obj.class.fullName);
        }
    });

    // Manage a snapshot manually
    const snapshot = Il2Cpp.MemorySnapshot.capture();
    // ... use snapshot.classes / snapshot.objects
    snapshot.free(); // must be freed manually
});
```

## Type Utilities

```javascript
Il2Cpp.perform(() => {
    // Box primitive types
    const boxedInt = Il2Cpp.boxed(42);
    const boxedBool = Il2Cpp.boxed(true);
    const boxedPtr = Il2Cpp.boxed(ptr(0xdeadbeef), "intptr");

    // Nullable value types
    const nullableInt = Il2Cpp.nullable(42);
    const nullValue = Il2Cpp.nullable(null, Il2Cpp.corlib.class("System.Int32"));

    // Type filters
    const IComparable = Il2Cpp.corlib.class("System.IComparable");
    const objects = [/* ... Il2Cpp.Object[] ... */];

    // Filter by type (including subclasses)
    const comparables = objects.filter(Il2Cpp.is(IComparable));

    // Exact type match
    const StringClass = Il2Cpp.corlib.class("System.String");
    const strings = objects.filter(Il2Cpp.isExactly(StringClass));

    // Allocate / free memory
    const ptr = Il2Cpp.alloc(64);
    // ... use it
    Il2Cpp.free(ptr);
});
```

        console.log("Thread:", t.id);

    // Schedule execution on the main thread
    Il2Cpp.perform(() => {
        // This code runs on the main thread
    }, "main");

## Complete Examples

### Modifying player attributes

```javascript
Il2Cpp.perform(() => {
    const playerClass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.Player");

    // Find every Player instance
    const players = Il2Cpp.gc.choose(playerClass);
    for (const player of players) {
        player.field("_health").value = 99999;
        player.field("_attack").value = 99999;
        player.field("_defense").value = 99999;
        console.log("Modified player:", player.method("get_Name").invoke());
    }
});
```

### Hooking a method and changing its return value

```javascript
Il2Cpp.perform(() => {
    const shopClass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.ShopManager");

    shopClass.method("GetPrice").implementation = function (itemId) {
        console.log("Getting item price, itemId =", itemId);
        return 0; // Free
    };

    shopClass.method("CanAfford").implementation = function (price) {
        return true; // Always affordable
    };
});
```

### Enumerating and searching classes

```javascript
Il2Cpp.perform(() => {
    // Search for classes whose name contains "Coin"
    for (const assembly of Il2Cpp.domain.assemblies) {
        for (const klass of assembly.image.classes) {
            if (klass.fullName.includes("Coin")) {
                console.log("Found:", klass.fullName);
                console.log(klass.toString()); // Print the full class structure
            }
        }
    }
});
```

## The flag Parameter of perform()

| Flag | Description |
|------|-------------|
| `"bind"` | (default) Attaches the thread and detaches automatically when the script is unloaded |
| `"free"` | Attaches the thread and detaches as soon as the callback finishes |
| `"leak"` | Attaches the thread and never detaches (leaks) |
| `"main"` | Schedules execution on the main thread |

```javascript
// Default "bind" mode
Il2Cpp.perform(() => { /* ... */ });

// Main-thread mode
Il2Cpp.perform(() => {
    // Runs on the Unity main thread
}, "main");
```
