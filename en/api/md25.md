# Il2Cpp (frida-il2cpp-bridge)

Integrates [frida-il2cpp-bridge](https://github.com/vfsfitvnm/frida-il2cpp-bridge) v0.13.0, providing the `globalThis.Il2Cpp` namespace for runtime reverse engineering of Unity Il2Cpp games/applications.

Supports Unity 5.3.0 ~ 6000.3.x. Only works in Unity applications running the Il2Cpp backend.

---

## Quick Start

```javascript
Il2Cpp.perform(() => {
    console.log("Unity version:", Il2Cpp.unityVersion);
    console.log("App identifier:", Il2Cpp.application.identifier);
    console.log("App version:", Il2Cpp.application.version);

    for (const assembly of Il2Cpp.domain.assemblies) {
        console.log("Assembly:", assembly.name);
    }
});
```

> `Il2Cpp.perform()` is the entry point for all operations. It automatically waits for `libil2cpp.so` to load and initialize, then attaches the current thread to the Il2Cpp domain.

---

## Configuration

Override auto-detected values via `Il2Cpp.$config` before calling `perform()`:

```javascript
Il2Cpp.$config.moduleName = "libil2cpp.so";
Il2Cpp.$config.unityVersion = "2021.3.0f1";
Il2Cpp.$config.exports = {
    il2cpp_class_from_name: () => Il2Cpp.module.base.add(0x1204c),
};

Il2Cpp.perform(() => { /* ... */ });
```

## Core Properties

| Property | Type | Description |
|----------|------|-------------|
| `Il2Cpp.module` | `Module` | Il2Cpp native module (libil2cpp.so) |
| `Il2Cpp.domain` | `Il2Cpp.Domain` | Application domain |
| `Il2Cpp.unityVersion` | `string` | Unity engine version |
| `Il2Cpp.application.identifier` | `string` | App package name |
| `Il2Cpp.application.version` | `string` | App version |
| `Il2Cpp.application.dataPath` | `string` | App data path |
| `Il2Cpp.corlib` | `Il2Cpp.Image` | Core library (mscorlib) |

## Class

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");

    // Info
    console.log("Name:", klass.name);
    console.log("Namespace:", klass.namespace);
    console.log("Full name:", klass.fullName);
    console.log("Instance size:", klass.instanceSize);
    console.log("Parent:", klass.parent?.fullName);

    // Type checks
    console.log("Abstract:", klass.isAbstract);
    console.log("Enum:", klass.isEnum);
    console.log("Interface:", klass.isInterface);
    console.log("Generic:", klass.isGeneric);
    console.log("Value type:", klass.isValueType);

    // Fields
    for (const field of klass.fields) {
        console.log(`  Field: ${field.name} (${field.type.name}) offset=0x${field.offset.toString(16)}`);
    }

    // Methods
    for (const method of klass.methods) {
        console.log(`  Method: ${method.name}(${method.parameterCount} params) -> ${method.returnType.name}`);
    }

    // Nested classes, interfaces, hierarchy
    for (const nested of klass.nestedClasses) { console.log("Nested:", nested.name); }
    for (const iface of klass.interfaces) { console.log("Interface:", iface.fullName); }
    for (const parent of klass.hierarchy()) { console.log("Inherits:", parent.fullName); }

    // Lookup by name
    const hpField = klass.field("hp");
    const attackMethod = klass.method("Attack", 1);

    // Create objects
    klass.initialize();
    const newObj = klass.new();

    // Enumerate all classes
    Il2Cpp.Class.enumerate((klass) => { /* callback */ });
});
```

## Method

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");
    const method = klass.method("TakeDamage", 1);

    // Info
    console.log("Name:", method.name);
    console.log("Params:", method.parameterCount);
    console.log("Return type:", method.returnType.name);
    console.log("Static:", method.isStatic);
    console.log("VA:", method.virtualAddress);

    // Invoke static method
    const result = klass.method("GetInstance").invoke();

    // Replace implementation
    method.implementation = function (damage) {
        console.log("TakeDamage called, damage =", damage);
        return;
    };

    // Revert to original
    method.revert();

    // Intercept (observe without replacing)
    method.intercept({
        onEnter(instance, parameters) {
            console.log("Enter TakeDamage");
        },
        onLeave(returnValue) {
            console.log("Leave TakeDamage");
        }
    });
});
```

## Object

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.Player");

    // Search heap for instances
    const objects = Il2Cpp.gc.choose(klass);
    for (const obj of objects) {
        // Instance field
        const hp = obj.field("hp");
        console.log("HP:", hp.value);
        hp.value = 99999;

        // Instance method
        const name = obj.method("get_Name").invoke();
        console.log("Name:", name);
    }
});
```

## Trace

```javascript
Il2Cpp.perform(() => {
    // Trace assembly methods with filters
    Il2Cpp.trace()
        .assemblies(Il2Cpp.domain.assembly("Assembly-CSharp"))
        .filterClasses(klass => klass.namespace === "Game")
        .filterMethods(method => !method.isStatic)
        .and()
        .attach();

    // Trace with parameter logging
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");
    Il2Cpp.trace(true)
        .methods(klass.method("TakeDamage"))
        .and()
        .attach();

    // Backtrace
    Il2Cpp.backtrace()
        .methods(klass.method("TakeDamage"))
        .and()
        .attach();
});
```

## Type Utilities

```javascript
Il2Cpp.perform(() => {
    // Boxing
    const boxedInt = Il2Cpp.boxed(42);

    // Nullable
    const nullableInt = Il2Cpp.nullable(42);

    // Type filters
    const IComparable = Il2Cpp.corlib.class("System.IComparable");
    const comparables = objects.filter(Il2Cpp.is(IComparable));
    const strings = objects.filter(Il2Cpp.isExactly(StringClass));

    // Il2Cpp string
    const str = Il2Cpp.string("Hello");

    // Memory
    const p = Il2Cpp.alloc(64);
    Il2Cpp.free(p);
});
```

---

## perform() Flags

| Flag | Description |
|------|-------------|
| `"bind"` | (default) Attach thread, auto-detach on script unload |
| `"free"` | Attach thread, detach when callback ends |
| `"leak"` | Attach thread, never detach (leak) |
| `"main"` | Schedule on Unity main thread |

## Full Example

```javascript
Il2Cpp.perform(() => {
    const playerClass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.Player");

    const players = Il2Cpp.gc.choose(playerClass);
    for (const player of players) {
        player.field("_health").value = 99999;
        player.field("_attack").value = 99999;
    }

    playerClass.method("GetPrice").implementation = function (itemId) {
        return 0; // Free
    };
});
```
