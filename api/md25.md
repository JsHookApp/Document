# Il2Cpp (frida-il2cpp-bridge)

内置集成了 [frida-il2cpp-bridge](https://github.com/vfsfitvnm/frida-il2cpp-bridge) v0.13.0，提供 `globalThis.Il2Cpp` 命名空间，用于对 Unity Il2Cpp 游戏/应用进行运行时逆向分析。

支持 Unity 5.3.0 ~ 6000.3.x，仅在运行 Il2Cpp 后端的 Unity 应用中有效。

---

## 快速开始

```javascript
Il2Cpp.perform(() => {
    // Il2Cpp 已初始化，可以安全访问所有 API
    console.log("Unity 版本:", Il2Cpp.unityVersion);
    console.log("应用标识:", Il2Cpp.application.identifier);
    console.log("应用版本:", Il2Cpp.application.version);

    // 遍历程序集
    for (const assembly of Il2Cpp.domain.assemblies) {
        console.log("程序集:", assembly.name);
    }
});
```

> `Il2Cpp.perform()` 是所有操作的入口点。它会自动等待 `libil2cpp.so` 加载并初始化完毕，然后附加当前线程到 Il2Cpp 域。

---

## 配置

可以通过 `Il2Cpp.$config` 在 `perform()` 调用前覆盖自动检测的值：

```javascript
// 覆盖模块名（默认 Android 上为 libil2cpp.so）
Il2Cpp.$config.moduleName = "libil2cpp.so";

// 覆盖 Unity 版本
Il2Cpp.$config.unityVersion = "2021.3.0f1";

// 覆盖导出函数地址（对抗混淆/剥离）
Il2Cpp.$config.exports = {
    il2cpp_class_from_name: () => Il2Cpp.module.base.add(0x1204c),
};

Il2Cpp.perform(() => { /* ... */ });
```

## 核心属性

| 属性 | 类型 | 说明 |
|------|------|------|
| `Il2Cpp.module` | `Module` | Il2Cpp 原生模块（libil2cpp.so） |
| `Il2Cpp.domain` | `Il2Cpp.Domain` | 应用程序域 |
| `Il2Cpp.unityVersion` | `string` | Unity 引擎版本 |
| `Il2Cpp.application.identifier` | `string` | 应用包名 |
| `Il2Cpp.application.version` | `string` | 应用版本 |
| `Il2Cpp.application.dataPath` | `string` | 应用数据路径 |
| `Il2Cpp.corlib` | `Il2Cpp.Image` | 核心库（mscorlib） |

## Domain（域）

```javascript
Il2Cpp.perform(() => {

    // 获取所有程序集
    const assemblies = domain.assemblies;

    // 按名称查找程序集

    // 附加线程
```

## Class（类）

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");

    // 基本信息
    console.log("名称:", klass.name);
    console.log("命名空间:", klass.namespace);
    console.log("全名:", klass.fullName);
    console.log("实例大小:", klass.instanceSize);
    console.log("父类:", klass.parent?.fullName);

    // 类型判断
    console.log("是否抽象:", klass.isAbstract);
    console.log("是否枚举:", klass.isEnum);
    console.log("是否接口:", klass.isInterface);
    console.log("是否泛型:", klass.isGeneric);
    console.log("是否值类型:", klass.isValueType);

    // 字段
    for (const field of klass.fields) {
        console.log(`  字段: ${field.name} (${field.type.name}) offset=0x${field.offset.toString(16)}`);
    }

    // 方法
    for (const method of klass.methods) {
        console.log(`  方法: ${method.name}(${method.parameterCount}个参数) -> ${method.returnType.name}`);
    }

    // 嵌套类
    for (const nested of klass.nestedClasses) {
        console.log(`  嵌套类: ${nested.name}`);
    }

    // 接口
    for (const iface of klass.interfaces) {
        console.log(`  接口: ${iface.fullName}`);
    }

    // 继承链
    for (const parent of klass.hierarchy()) {
        console.log(`  继承: ${parent.fullName}`);
    }

    // 按名称获取字段/方法
    const hpField = klass.field("hp");
    const attackMethod = klass.method("Attack", 1); // 1个参数

    // 初始化静态构造函数
    klass.initialize();

    // 创建新对象
    const newObj = klass.new();

    // 枚举所有已定义的类
    Il2Cpp.Class.enumerate((klass) => {
        // 每个类的回调
    });
});
```

## Method（方法）

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");

    const method = klass.method("TakeDamage", 1);

    // 基本信息
    console.log("名称:", method.name);
    console.log("参数数:", method.parameterCount);
    console.log("返回类型:", method.returnType.name);
    console.log("是否静态:", method.isStatic);
    console.log("访问修饰符:", method.modifier);
    console.log("虚拟地址:", method.virtualAddress);
    console.log("RVA:", method.relativeVirtualAddress);

    // 参数信息
    for (const param of method.parameters) {
        console.log(`  参数: ${param.name} (${param.type.name})`);
    }

    // 调用静态方法
    const staticMethod = klass.method("GetInstance");
    const result = staticMethod.invoke();

    // 替换方法实现
    method.implementation = function (damage) {
        console.log("TakeDamage 被调用, damage =", damage);
        // 返回原始行为或自定义值
        return;
    };

    // 恢复原始实现
    method.revert();

    // 拦截方法（不替换，只监听）
    method.intercept({
        onEnter(instance, parameters) {
            console.log("进入 TakeDamage, this =", instance, "damage =", parameters.damage);
        },
        onLeave(returnValue) {
            console.log("离开 TakeDamage, 返回 =", returnValue);
        }
    });

    // 泛型方法实例化
    const genericMethod = klass.method("DoSomething");
    const inflated = genericMethod.inflate(Il2Cpp.corlib.class("System.Int32"));
});
```

## Object（对象）

```javascript
Il2Cpp.perform(() => {
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");

    // 通过 GC 搜索堆中对象
    const objects = Il2Cpp.gc.choose(klass);
    for (const obj of objects) {
        // 实例字段 — 通过对象访问
        const hp = obj.field("hp");
        console.log("当前 HP:", hp.value);
        hp.value = 99999;

        // 调用实例方法
        const name = obj.method("get_Name").invoke();
        console.log("玩家名:", name);

        // 访问基类成员
        const baseObj = obj.base;
        console.log("基类:", baseObj.class.fullName);
    }

    // 创建新对象
    const newObj = klass.new();

    // 分配（不调用构造函数）
    const allocated = klass.alloc();
});
```

## Dump（类转储）

```javascript
Il2Cpp.perform(() => {

    // 默认路径: /data/data/<包名>/files/<包名>_<版本>.cs

    Il2Cpp.dumpTree("/data/local/tmp/dump");
});
```

## Trace（方法追踪）

```javascript
Il2Cpp.perform(() => {
    // 追踪整个程序集中的方法
    Il2Cpp.trace()
        .assemblies(Il2Cpp.domain.assembly("Assembly-CSharp"))
        .filterClasses(klass => klass.namespace === "Game")
        .filterMethods(method => !method.isStatic)
        .and()
        .attach();

    // 追踪特定类的所有方法
    const klass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.PlayerController");
    Il2Cpp.trace()
        .classes(klass)
        .and()
        .attach();

    // 追踪特定方法（含参数值）
    Il2Cpp.trace(true) // true = 记录参数
        .methods(klass.method("TakeDamage"))
        .and()
        .attach();

    // 带调用栈追踪
    Il2Cpp.backtrace()
        .methods(klass.method("TakeDamage"))
        .and()
        .attach();

    // 详细模式（不去重）
    Il2Cpp.trace()
        .verbose(true)
        .domain()
        .filterAssemblies(a => a.name === "Assembly-CSharp")
        .filterClasses(c => c.fullName.startsWith("Game."))
        .and()
        .attach();
});
```

## GC（垃圾回收）

```javascript
Il2Cpp.perform(() => {
    // GC 信息
    console.log("堆大小:", Il2Cpp.gc.heapSize);
    console.log("已使用:", Il2Cpp.gc.usedHeapSize);
    console.log("是否启用:", Il2Cpp.gc.isEnabled);
    console.log("是否增量:", Il2Cpp.gc.isIncremental);

    // 禁用/启用 GC
    Il2Cpp.gc.isEnabled = false;
    Il2Cpp.gc.isEnabled = true;

    // 搜索堆中指定类的实例
    const playerClass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.Player");
    const players = Il2Cpp.gc.choose(playerClass);
    console.log("找到", players.length, "个 Player 实例");
});
```

## Memory Snapshot（内存快照）

```javascript
Il2Cpp.perform(() => {
    // 使用内存快照（自动释放）
    Il2Cpp.memorySnapshot(snapshot => {
        // 所有已初始化的类
        for (const klass of snapshot.classes) {
            console.log(klass.fullName);
        }

        // 所有被追踪的对象
        for (const obj of snapshot.objects) {
            console.log(obj.class.fullName);
        }
    });

    // 手动管理快照
    const snapshot = Il2Cpp.MemorySnapshot.capture();
    // ... 使用 snapshot.classes / snapshot.objects
    snapshot.free(); // 必须手动释放
});
```

## 类型工具

```javascript
Il2Cpp.perform(() => {
    // 装箱基本类型
    const boxedInt = Il2Cpp.boxed(42);
    const boxedBool = Il2Cpp.boxed(true);
    const boxedPtr = Il2Cpp.boxed(ptr(0xdeadbeef), "intptr");

    // 可空值类型
    const nullableInt = Il2Cpp.nullable(42);
    const nullValue = Il2Cpp.nullable(null, Il2Cpp.corlib.class("System.Int32"));

    // 类型过滤器
    const IComparable = Il2Cpp.corlib.class("System.IComparable");
    const objects = [/* ... Il2Cpp.Object[] ... */];

    // 按类型过滤（包含子类）
    const comparables = objects.filter(Il2Cpp.is(IComparable));

    // 精确类型匹配
    const StringClass = Il2Cpp.corlib.class("System.String");
    const strings = objects.filter(Il2Cpp.isExactly(StringClass));

    // 内存分配/释放
    const ptr = Il2Cpp.alloc(64);
    // ... 使用
    Il2Cpp.free(ptr);
});
```

## Thread（线程）

```javascript
Il2Cpp.perform(() => {
    // 当前线程
    const current = Il2Cpp.currentThread;
    console.log("线程 ID:", current.id);
    console.log("托管 ID:", current.managedId);
    console.log("是否终结器:", current.isFinalizer);

    // 主线程
    const main = Il2Cpp.mainThread;

    // 所有线程
    const threads = Il2Cpp.attachedThreads;
    for (const t of threads) {
        console.log("线程:", t.id);
    }

    // 在主线程调度执行
    Il2Cpp.perform(() => {
        // 这段代码在主线程执行
    }, "main");
});
```

## 完整示例

### 修改玩家属性

```javascript
Il2Cpp.perform(() => {
    const playerClass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.Player");

    // 搜索所有 Player 实例
    const players = Il2Cpp.gc.choose(playerClass);
    for (const player of players) {
        player.field("_health").value = 99999;
        player.field("_attack").value = 99999;
        player.field("_defense").value = 99999;
        console.log("已修改玩家:", player.method("get_Name").invoke());
    }
});
```

### Hook 方法并修改返回值

```javascript
Il2Cpp.perform(() => {
    const shopClass = Il2Cpp.domain.assembly("Assembly-CSharp")
        .image.class("Game.ShopManager");

    shopClass.method("GetPrice").implementation = function (itemId) {
        console.log("获取物品价格, itemId =", itemId);
        return 0; // 免费
    };

    shopClass.method("CanAfford").implementation = function (price) {
        return true; // 总是买得起
    };
});
```

### 枚举并搜索类

```javascript
Il2Cpp.perform(() => {
    // 搜索包含 "Coin" 的类
    for (const assembly of Il2Cpp.domain.assemblies) {
        for (const klass of assembly.image.classes) {
            if (klass.fullName.includes("Coin")) {
                console.log("找到:", klass.fullName);
                console.log(klass.toString()); // 打印完整类结构
            }
        }
    }
});
```

## perform() 的 flag 参数

| Flag | 说明 |
|------|------|
| `"bind"` | (默认) 附加线程，脚本卸载时自动分离 |
| `"free"` | 附加线程，回调结束后立即分离 |
| `"leak"` | 附加线程，不自动分离（泄漏） |
| `"main"` | 在主线程上调度执行 |

```javascript
// 默认 bind 模式
Il2Cpp.perform(() => { /* ... */ });

// 主线程模式
Il2Cpp.perform(() => {
    // 在 Unity 主线程执行
}, "main");
```
